### Сортировка вставкой
```c
void ins_sort(int a[], int n) {
	int i, j, tmp;
	for (j=1; j<n; j++ a[i+1]=tmp) {
		for (tmp=a[j], i=j-1; i>=0 && tmp<a[i]; a[i+1]=a[i],i--);
	}
}
```

```c
void quick_sort(int a[], int x, int y) {
	int i, j, tmp, f;
	f=0;
	i=x;
	j=y;
	while (i<j){
		if (a[i]>a[j]){
			tmp=a[i];
			a[i]=a[j];
			a[j]=tmp;
			f=!f;
		}
		if(f) i++;
		else j--;
	}
	if (x<i-1){
		quick_sort(a, x, i-1);
	}
	if (x+1<y){
		quick_sort(a, i+1, y);
	}
}
```
### Четная-нечетная сортировка
```c
int isswap(int a, int b){
	if (!(a%2)&&b%2) return 1;
	if (!(a%2)&& !(b%2) && a>b)
		return 1;
	if(a%2 && b%2 && a<b)
		return 1;
	return 0;
}

void oddeven_sort(int a[5], int n){
	int i,j,m,tmp;
	for (j=0; j<n-1; j++){
		for(m=j, i=j+1; i<n; i++)
		if (isswap(a[i], a[m]))
			m=i;
	tmp=a[j];
	a[j]=a[m];
	a[m]=tmp;
	}
}
```
### Удалить все вхождения минимального элемента в массиве
```c
int del_min(int a[], int n){
    int i,j,m;
    for (m=a[0], i=1; i<n; i++){
        if(a[i]<m) m=a[i];
    }
    for (i=0, j=0; i<n; i++){
        if(a[i] != m) a[j++]=a[i];
    }
    return j;
}
```
